Core Agent Architecture

import openai
from enum import Enum

class IncidentSeverity(Enum):
    CRITICAL = "critical"
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"

class AIOnCallAgent:
    def __init__(self, openai_key, confidence_threshold=0.85):
        openai.api_key = openai_key
        self.confidence_threshold = confidence_threshold
        self.remediation_history = []
        
    def handle_incident(self, alert_data):
        """Main incident handling workflow"""
        
        # Step 1: Triage
        triage = self._triage_incident(alert_data)
        
        # Step 2: Check if AI can handle
        if triage['confidence'] < self.confidence_threshold:
            return self._escalate_to_human(alert_data, triage)
        
        # Step 3: Get remediation plan
        plan = self._generate_remediation_plan(alert_data, triage)
        
        # Step 4: Execute with safety checks
        result = self._execute_remediation(plan, alert_data)
        
        # Step 5: Verify resolution
        if result['success']:
            self._verify_resolution(alert_data)
            self._log_success(alert_data, plan, result)
            return {'status': 'resolved', 'method': 'ai'}
        else:
            return self._escalate_to_human(alert_data, triage, failed_attempt=result)
    
    def _triage_incident(self, alert_data):
        """Use LLM to classify and assess incident"""
        
        prompt = f"""Analyze this production alert and provide triage assessment:

ALERT: {alert_data['title']}
SERVICE: {alert_data['service']}
SEVERITY: {alert_data['severity']}
METRICS: {alert_data['metrics']}
RECENT LOGS: {alert_data['logs'][:500]}

Provide JSON response:
{{
  "incident_type": "service_down|high_latency|disk_full|memory_leak|other",
  "root_cause_hypothesis": "brief description",
  "can_auto_remediate": true/false,
  "confidence": 0.0-1.0,
  "recommended_action": "specific action",
  "risk_level": "low|medium|high",
  "reasoning": "why this assessment"
}}"""

        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are an expert SRE triaging production incidents. Be conservative - only recommend auto-remediation if you're highly confident."},
                {"role": "user", "content": prompt}
            ],
            temperature=0.1
        )
        
        import json
        return json.loads(response.choices[0].message.content)
    
    def _generate_remediation_plan(self, alert_data, triage):
        """Generate step-by-step remediation plan"""
        
        prompt = f"""Generate safe remediation plan for this incident:

INCIDENT TYPE: {triage['incident_type']}
ROOT CAUSE: {triage['root_cause_hypothesis']}
SERVICE: {alert_data['service']}

Generate JSON plan:
{{
  "steps": [
    {{
      "action": "specific command or API call",
      "description": "what this does",
      "rollback": "how to undo if it fails",
      "safety_check": "what to verify before proceeding"
    }}
  ],
  "estimated_duration": "minutes",
  "rollback_plan": "full rollback procedure"
}}

SAFETY RULES:
- Never delete data
- Always have rollback
- Verify before each step
- Stop if anything unexpected"""

        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are generating safe, production-ready remediation plans. Safety is paramount."},
                {"role": "user", "content": prompt}
            ],
            temperature=0.1
        )
        
        import json
        return json.loads(response.choices[0].message.content)
    
    def _execute_remediation(self, plan, alert_data):
        """Execute remediation with safety checks"""
        
        executed_steps = []
        
        for step in plan['steps']:
            # Safety check before execution
            if not self._safety_check(step, alert_data):
                return {
                    'success': False,
                    'reason': 'Safety check failed',
                    'executed_steps': executed_steps
                }
            
            # Execute step
            try:
                result = self._execute_step(step, alert_data)
                executed_steps.append({
                    'step': step,
                    'result': result,
                    'success': True
                })
                
                # Verify step succeeded
                if not self._verify_step(step, result, alert_data):
                    # Rollback
                    self._rollback(executed_steps)
                    return {
                        'success': False,
                        'reason': 'Step verification failed',
                        'executed_steps': executed_steps
                    }
                    
            except Exception as e:
                # Rollback on error
                self._rollback(executed_steps)
                return {
                    'success': False,
                    'reason': str(e),
                    'executed_steps': executed_steps
                }
        
        return {
            'success': True,
            'executed_steps': executed_steps
        }
    
    def _safety_check(self, step, alert_data):
        """Verify it's safe to execute this step"""
        
        # Check 1: Is this a known safe operation?
        safe_operations = [
            'restart_service',
            'clear_cache',
            'increase_connection_pool',
            'scale_up_instances',
            'clear_disk_space'
        ]
        
        if step['action'] not in safe_operations:
            return False
        
        # Check 2: Is service in maintenance window?
        if self._is_maintenance_window():
            return True
        
        # Check 3: Is this a critical service?
        if alert_data.get('service_tier') == 'critical':
            # Extra caution for critical services
            return step.get('risk_level') == 'low'
        
        return True
    
    def _execute_step(self, step, alert_data):
        """Execute a single remediation step"""
        
        action = step['action']
        service = alert_data['service']
        
        if action == 'restart_service':
            return self._restart_service(service)
        elif action == 'clear_cache':
            return self._clear_cache(service)
        elif action == 'increase_connection_pool':
            return self._increase_connection_pool(service)
        elif action == 'scale_up_instances':
            return self._scale_up(service)
        elif action == 'clear_disk_space':
            return self._clear_disk_space(service)
        else:
            raise Exception(f"Unknown action: {action}")
    
    def _restart_service(self, service):
        """Restart service using ECS/Kubernetes"""
        import boto3
        
        ecs = boto3.client('ecs')
        
        # Force new deployment (rolling restart)
        response = ecs.update_service(
            cluster='production',
            service=service,
            forceNewDeployment=True
        )
        
        return {'status': 'restarting', 'deployment_id': response['service']['deployments'][0]['id']}
    
    def _verify_resolution(self, alert_data):
        """Verify incident is actually resolved"""
        
        # Wait 2 minutes for metrics to stabilize
        import time
        time.sleep(120)
        
        # Check metrics
        current_metrics = fetch_current_metrics(alert_data['service'])
        
        # Verify metrics are back to normal
        if current_metrics['error_rate'] < 1.0 and current_metrics['latency_p99'] < 1000:
            return True
        
        return False
    
    def _escalate_to_human(self, alert_data, triage, failed_attempt=None):
        """Escalate to on-call engineer"""
        
        context = {
            'alert': alert_data,
            'ai_triage': triage,
            'failed_attempt': failed_attempt
        }
        
        # Page engineer

            severity=alert_data['severity'],
            message=f"AI escalation: {alert_data['title']}",

        )
        
        return {'status': 'escalated', 'method': 'human'}

Safety Guardrails

What AI Is NOT Allowed To Do

  • ❌ Delete production data
  • ❌ Modify database schemas
  • ❌ Change security groups or firewall rules
  • ❌ Deploy code changes
  • ❌ Modify IAM permissions
  • ❌ Take actions during security incidents
  • ❌ Make decisions affecting customer data

Required Safety Checks

  • ✅ Confidence threshold > 85%
  • ✅ Rollback plan for every action
  • ✅ Verification after each step
  • ✅ Human approval for critical services
  • ✅ Automatic escalation if anything unexpected
  • ✅ Complete audit log of all actions
